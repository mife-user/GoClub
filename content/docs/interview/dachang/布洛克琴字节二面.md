## 算法

- 现有学生表（Students）和科目成绩表（Scores）。请编写一个 SQL 查询，统计总分严格大于 300 分的学生人数。要求在处理过程中，先得出每位学生的各科总成绩并按总分从大到小进行排序，最终输出符合上述条件的统计结果。
- 答案
    
    ```jsx
    SELECT COUNT(*) AS student_count
    FROM (
        -- 1. 关联表并按学生分组计算总分
        SELECT s.student_id
        FROM Students s
        JOIN Scores sc ON s.student_id = sc.student_id
        GROUP BY s.student_id
        -- 2. 筛选总分大于 300 的记录
        HAVING SUM(sc.score) > 300
        -- 3. 按总分从大到小排序
        ORDER BY SUM(sc.score) DESC
    ) AS result_table;
    ```
    
- 三个线程分别打印a、b、c，多线程打印abc五遍
- 答案
    
    ```jsx
    package main
    
    import (
    	"fmt"
    )
    
    func a(ca, ab chan bool) {
    	for i := 0; i < 5; i++ {
    		<- ca
    		fmt.Print("a")
    		ab <- true
    	}
    }
    func b(ab, bc chan bool) {
    	for i := 0; i < 5; i++ {
    		<- ab
    		fmt.Print("b")
    		bc <- true
    	}
    }
    func c(bc,ca, done chan bool) {
    	for i := 0; i < 5; i++ {
    		<- bc
    		fmt.Println("c")
    		if i == 4 {
    			done <- true
    		} else {
    			ca <- true
    		}
    	}
    }
    
    func main() {
    	ab := make(chan bool)
    	bc := make(chan bool)
    	ca := make(chan bool)
    	done := make(chan bool)
    	go a(ca, ab)
    	go b(ab, bc)
    	go c(bc, ca, done)
    	ca <- true
    	<- done
    }
    
    ```
